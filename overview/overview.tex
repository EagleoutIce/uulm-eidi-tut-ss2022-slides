    \pgfmathsetmacro\diff{\curXInit-\blockX}
    \ifdim\diff pt>.1pt\relax
        % only give it a chance to subtract
        \pgfmathsetmacro\toss{random(1,3)}%
        \ifnum\toss=1 % we increase dec chance
            \let\curX\curXInit
        \else
            \pgfmathsetmacro\curX{\curXInit-\blockcodeindent}
        \fi
    \else\ifdim\diff pt<0pt\relax% only inc
        \pgfmathsetmacro\toss{random(1,2)}%
        \ifnum\toss=1
            \pgfmathsetmacro\curX{\curXInit+\blockcodeindent}
        \else
            \let\curX\curXInit
        \fi
    \else
        \pgfmathsetmacro\toss{random(1,3)}%
        \ifnum\toss=1
            \pgfmathsetmacro\curX{\curXInit-\blockcodeindent}
        \else\ifnum\toss=2
            \pgfmathsetmacro\curX{\curXInit+\blockcodeindent}
        \else\let\curX\curXInit
        \fi\fi
    \fi\fi
    \ifnum\i=1 \ifnum\line=0 \let\curX\blockX \fi\fi % not for the first line in the first block
